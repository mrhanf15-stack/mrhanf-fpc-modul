# ============================================================
# Mr. Hanf Full Page Cache v8.0.4 - .htaccess Regeln
# ============================================================
# Architektur: Request -> fpc_serve.php (PHP) -> readfile()
#
# HINWEIS: Die direkte Apache-Auslieferung (cache/fpc/index.html)
# verursacht auf Artfiles Shared Hosting einen Redirect-Loop,
# weil die CLEAN SEO URL Regel (-f Pruefung + [PT]) die
# umgeschriebene URL erneut verarbeitet. Deshalb wird
# fpc_serve.php als Handler verwendet, das die Cache-Datei
# per readfile() ausliefert und mit exit beendet.
#
# fpc_serve.php Overhead: ~5ms (validiert + readfile + exit)
# Ohne FPC: 500-5000ms (PHP-Backend generiert Seite live)
#
# Sicherheit:
#   - fpc_serve.php validiert: Dateigroesse, TTL, Health-Marker
#   - Kein Directory Traversal moeglich (realpath-Check)
#   - Korrupte/leere Dateien werden automatisch geloescht
#   - Cache-Dateien haben FPC-VALID Marker + Mindestgroesse
#
# CHANGELOG v8.0.4:
#   - FIX: Redirect-Loop endgueltig behoben!
#     Ursache: Apache [L] Flag startet Regelverarbeitung neu,
#     CLEAN SEO URL (-f Pruefung + [PT]) erkennt fpc_serve.php
#     als existierende Datei und leitet erneut weiter -> Loop.
#     Loesung: [END] statt [L] stoppt Regelverarbeitung komplett.
#   - CHANGE: Zurueck zu fpc_serve.php statt direkter HTML-Auslieferung
#     (Artfiles cache/.htaccess blockiert .html mit 403 + Redirect-Loop)
#   - WICHTIG: In CLEAN SEO URL muessen diese Ausschluesse stehen:
#     1. RewriteCond %{REQUEST_URI} !^/cache/fpc/ [NC]
#        (vor der index.html Redirect-Regel)
#     2. RewriteCond %{REQUEST_URI} !^/cache/fpc/ [NC]
#        RewriteCond %{REQUEST_URI} !^/fpc_serve\.php [NC]
#        (vor der -f/-d Pruefung)
#
# CHANGELOG v8.0.3:
#   - FIX: Cache-Control Header-Konflikt behoben (always set)
#   - NEU: Preloader cached jetzt auch Kategorie-Seiten (aus DB)
#
# CHANGELOG v8.0.2:
#   - FIX: FPC-Header per env-Variable statt FilesMatch
#
# CHANGELOG v8.0.1:
#   - NEU: /kasse und /warenkorb zur Ausschlussliste
#
# CHANGELOG v8.0.0:
#   - Erste v8.0 Version
#
# INSTALLATION:
#   1. Diesen Block am ANFANG der .htaccess einfuegen (vor allem anderen)
#   2. In CLEAN SEO URL die Ausschluesse einfuegen (siehe oben)
#   3. fpc_serve.php ins Shop-Root kopieren
#   4. cache/fpc/ Verzeichnis anlegen (chmod 777)
# ============================================================

# --- FPC START v8.0 ---
<IfModule mod_rewrite.c>
RewriteEngine On

# === Startseite ===
RewriteCond %{REQUEST_METHOD} =GET
RewriteCond %{HTTP_COOKIE} !MODsid [NC]
RewriteCond %{HTTP_COOKIE} !PHPSESSID [NC]
RewriteCond %{REQUEST_URI} !^/admin [NC]
RewriteCond %{QUERY_STRING} ^$
RewriteCond %{DOCUMENT_ROOT}/cache/fpc/index.html -f
RewriteRule ^$ fpc_serve.php [END,QSA,E=FPC_HIT:1]

# === Unterseiten (ALLE Conds wiederholt!) ===
RewriteCond %{REQUEST_METHOD} =GET
RewriteCond %{HTTP_COOKIE} !MODsid [NC]
RewriteCond %{HTTP_COOKIE} !PHPSESSID [NC]
RewriteCond %{REQUEST_URI} !^/admin [NC]
RewriteCond %{REQUEST_URI} !\.(css|js|jpg|jpeg|png|gif|svg|ico|woff|woff2|ttf|eot|pdf|zip)$ [NC]
RewriteCond %{QUERY_STRING} ^$
RewriteCond %{REQUEST_URI} !checkout [NC]
RewriteCond %{REQUEST_URI} !^/kasse [NC]
RewriteCond %{REQUEST_URI} !login [NC]
RewriteCond %{REQUEST_URI} !account [NC]
RewriteCond %{REQUEST_URI} !shopping_cart [NC]
RewriteCond %{REQUEST_URI} !^/warenkorb [NC]
RewriteCond %{REQUEST_URI} !logoff [NC]
RewriteCond %{REQUEST_URI} !^/vergleich [NC]
RewriteCond %{REQUEST_URI} !^/wishlist [NC]
RewriteCond %{REQUEST_URI} !password_double_opt [NC]
RewriteCond %{REQUEST_URI} !create_account [NC]
RewriteCond %{REQUEST_URI} !contact_us [NC]
RewriteCond %{REQUEST_URI} !tell_a_friend [NC]
RewriteCond %{REQUEST_URI} !product_reviews_write [NC]
RewriteCond %{DOCUMENT_ROOT}/cache/fpc/%{REQUEST_URI}/index.html -f
RewriteRule ^(.+)$ fpc_serve.php [END,QSA,E=FPC_HIT:1]
</IfModule>

# === FPC Cache-Control Header (via Environment-Variable) ===
# "always set" statt "set" damit der FPC-Header NACH dem Shop-Header
# gesetzt wird und den FilesMatch .html -> max-age=1 ueberschreibt.
# HINWEIS: Bei [END] wird REDIRECT_FPC_HIT moeglicherweise nicht gesetzt.
# fpc_serve.php setzt die Header selbst per PHP header() Funktion.
# Diese Regeln hier sind ein zusaetzlicher Fallback.
<IfModule mod_headers.c>
    Header always set X-FPC-Cache "HIT" env=REDIRECT_FPC_HIT
    Header always set X-FPC-Version "8.0.4" env=REDIRECT_FPC_HIT
    Header always set Cache-Control "no-store, no-cache, must-revalidate" env=REDIRECT_FPC_HIT
    Header always set Pragma "no-cache" env=REDIRECT_FPC_HIT
</IfModule>
# --- FPC ENDE v8.0 ---
