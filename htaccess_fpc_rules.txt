# ============================================================
# Mr. Hanf Full Page Cache v8.0.1 - .htaccess Regeln
# ============================================================
# WICHTIG v8.0: Apache liefert gecachte Seiten DIREKT als
# statische HTML-Dateien aus - OHNE PHP-Worker!
#
# Vorher (v7.x): Request -> fpc_serve.php (PHP) -> readfile()
# Jetzt  (v8.0): Request -> Apache liefert index.html direkt
#
# Vorteile:
#   - Kein PHP-Worker belegt -> Server kann mehr Besucher bedienen
#   - Schneller als PHP (kein PHP-Startup, kein readfile())
#   - Weniger Worker-Spikes bei hohem Traffic
#
# Sicherheit:
#   - Validierung findet NUR beim SCHREIBEN statt (fpc_preloader.php)
#   - fpc_serve.php bleibt als Fallback-Option erhalten
#   - Cache-Dateien haben FPC-VALID Marker + Mindestgroesse
#
# CHANGELOG v8.0.1:
#   - NEU: /kasse zur Ausschlussliste (SEO-URL fuer Checkout/Payment)
#   - NEU: /warenkorb zur Ausschlussliste (SEO-URL fuer Warenkorb)
#   - Praeventive Absicherung: Kreditkarten-/Payment-Seiten explizit geschuetzt
#
# CHANGELOG v8.0.0:
#   - NEU: Direkte Apache-Auslieferung (kein PHP fuer gecachte Seiten)
#   - NEU: Content-Type Header via AddType
#   - NEU: Cache-Control Header via mod_headers
#   - ENTFERNT: Weiterleitung an fpc_serve.php
# ============================================================

# --- FPC START v8.0 ---
<IfModule mod_rewrite.c>
RewriteEngine On

# === Startseite ===
RewriteCond %{REQUEST_METHOD} =GET
RewriteCond %{HTTP_COOKIE} !MODsid [NC]
RewriteCond %{HTTP_COOKIE} !PHPSESSID [NC]
RewriteCond %{REQUEST_URI} !^/admin [NC]
RewriteCond %{QUERY_STRING} ^$
RewriteCond %{DOCUMENT_ROOT}/cache/fpc/index.html -f
RewriteRule ^$ cache/fpc/index.html [L,T=text/html]

# === Unterseiten (ALLE Conds wiederholt!) ===
RewriteCond %{REQUEST_METHOD} =GET
RewriteCond %{HTTP_COOKIE} !MODsid [NC]
RewriteCond %{HTTP_COOKIE} !PHPSESSID [NC]
RewriteCond %{REQUEST_URI} !^/admin [NC]
RewriteCond %{REQUEST_URI} !\.(css|js|jpg|jpeg|png|gif|svg|ico|woff|woff2|ttf|eot|pdf|zip)$ [NC]
RewriteCond %{QUERY_STRING} ^$
RewriteCond %{REQUEST_URI} !checkout [NC]
RewriteCond %{REQUEST_URI} !^/kasse [NC]
RewriteCond %{REQUEST_URI} !login [NC]
RewriteCond %{REQUEST_URI} !account [NC]
RewriteCond %{REQUEST_URI} !shopping_cart [NC]
RewriteCond %{REQUEST_URI} !^/warenkorb [NC]
RewriteCond %{REQUEST_URI} !logoff [NC]
RewriteCond %{REQUEST_URI} !^/vergleich [NC]
RewriteCond %{REQUEST_URI} !^/wishlist [NC]
RewriteCond %{REQUEST_URI} !password_double_opt [NC]
RewriteCond %{REQUEST_URI} !create_account [NC]
RewriteCond %{REQUEST_URI} !contact_us [NC]
RewriteCond %{REQUEST_URI} !tell_a_friend [NC]
RewriteCond %{REQUEST_URI} !product_reviews_write [NC]
RewriteCond %{DOCUMENT_ROOT}/cache/fpc/%{REQUEST_URI}/index.html -f
RewriteRule ^(.+)$ cache/fpc/$1/index.html [L,T=text/html]
</IfModule>

# === FPC Cache-Control Header ===
<IfModule mod_headers.c>
<FilesMatch "^cache/fpc/.+\.html$">
    Header set X-FPC-Cache "HIT"
    Header set X-FPC-Version "8.0.1"
    Header set Cache-Control "no-store, no-cache, must-revalidate"
    Header set Pragma "no-cache"
</FilesMatch>
</IfModule>
# --- FPC ENDE v8.0 ---
