# ============================================================
# Mr. Hanf Full Page Cache v8.0.7 - .htaccess Regeln
# ============================================================
# Architektur: Request -> fpc_serve.php (PHP) -> readfile()
#
# HINWEIS: Die direkte Apache-Auslieferung (cache/fpc/index.html)
# verursacht auf Artfiles Shared Hosting einen Redirect-Loop,
# weil die CLEAN SEO URL Regel (-f Pruefung + [PT]) die
# umgeschriebene URL erneut verarbeitet. Deshalb wird
# fpc_serve.php als Handler verwendet, das die Cache-Datei
# per readfile() ausliefert und mit exit beendet.
#
# fpc_serve.php Overhead: ~77ms (validiert + readfile + exit)
# Ohne FPC: 500-5000ms (PHP-Backend generiert Seite live)
#
# Sicherheit:
#   - fpc_serve.php validiert: Dateigroesse, TTL, Health-Marker
#   - Kein Directory Traversal moeglich (realpath-Check)
#   - Korrupte/leere Dateien werden automatisch geloescht
#   - Cache-Dateien haben FPC-VALID Marker + Mindestgroesse
#
# CHANGELOG v8.0.7:
#   - NEU: fpc_bypass Cookie-System fuer Warenkorb/Login
#     Autoinclude 95_fpc_bypass_cookie.php setzt "fpc_bypass=1" wenn
#     der Warenkorb gefuellt ist oder der Benutzer eingeloggt ist.
#     .htaccess und fpc_serve.php pruefen dieses Cookie.
#     Gastbesucher ohne Warenkorb bekommen weiterhin den FPC-Cache.
#
# CHANGELOG v8.0.6:
#   - FIX: Cookie-Check (MODsid/PHPSESSID) aus .htaccess UND fpc_serve.php entfernt!
#     modified-Shop setzt bei JEDEM Besucher (auch Gaeste) sofort ein MODsid-Cookie.
#     Dadurch wurde der FPC fuer ALLE Besucher blockiert.
#     Schutz erfolgt jetzt ueber URL-basierte Ausschlussliste in fpc_serve.php.
#   - FIX: %{DOCUMENT_ROOT} durch absoluten Pfad ersetzt.
#     Artfiles liefert /home/www/doc/28856/mr-hanf.de/www (Symlink) statt
#     /home/www/doc/28856/dcp288560004/mr-hanf.de/www (real). Der -f Check
#     schlaegt mit dem Symlink-Pfad fehl.
#   - FIX: [END] durch [L] ersetzt (bessere Kompatibilitaet mit Artfiles Apache)
#   - FIX: RewriteRule ^$ durch ^/?$ ersetzt (robuster fuer Startseite)
#
# CHANGELOG v8.0.5:
#   - FIX: Verzeichnis-Schutz in Preloader, Flush und Admin-Modul
#
# CHANGELOG v8.0.4:
#   - FIX: Redirect-Loop behoben (fpc_serve.php statt direkter HTML)
#
# CHANGELOG v8.0.3:
#   - FIX: Cache-Control Header-Konflikt behoben (always set)
#
# CHANGELOG v8.0.2:
#   - FIX: FPC-Header per env-Variable statt FilesMatch
#
# CHANGELOG v8.0.1:
#   - NEU: /kasse und /warenkorb zur Ausschlussliste
#
# CHANGELOG v8.0.0:
#   - Erste v8.0 Version
#
# INSTALLATION:
#   1. Diesen Block am ANFANG der .htaccess einfuegen (vor allem anderen)
#   2. WICHTIG: Den absoluten Pfad anpassen (siehe Zeilen mit -f Check)!
#      Ermitteln mit: php -r "echo __DIR__;"
#   3. In CLEAN SEO URL die Ausschluesse einfuegen:
#      - RewriteCond %{REQUEST_URI} !^/cache/fpc/ [NC]
#      - RewriteCond %{REQUEST_URI} !^/fpc_serve\.php [NC]
#   4. fpc_serve.php ins Shop-Root kopieren
#   5. cache/fpc/ Verzeichnis anlegen (chmod 777)
# ============================================================

# --- FPC START v8.0.7 ---
<IfModule mod_rewrite.c>
RewriteEngine On

# === Startseite ===
# v8.0.7: fpc_bypass Cookie statt MODsid pruefen.
# 95_fpc_bypass_cookie.php setzt "fpc_bypass=1" bei Warenkorb/Login.
RewriteCond %{REQUEST_METHOD} =GET
RewriteCond %{HTTP_COOKIE} !fpc_bypass [NC]
RewriteCond %{REQUEST_URI} !^/admin [NC]
RewriteCond %{QUERY_STRING} ^$
RewriteCond /home/www/doc/28856/dcp288560004/mr-hanf.de/www/cache/fpc/index.html -f
RewriteRule ^/?$ fpc_serve.php [L,QSA,E=FPC_HIT:1]

# === Unterseiten (ALLE Conds wiederholt!) ===
RewriteCond %{REQUEST_METHOD} =GET
RewriteCond %{HTTP_COOKIE} !fpc_bypass [NC]
RewriteCond %{REQUEST_URI} !^/admin [NC]
RewriteCond %{REQUEST_URI} !\.(css|js|jpg|jpeg|png|gif|svg|ico|woff|woff2|ttf|eot|pdf|zip)$ [NC]
RewriteCond %{QUERY_STRING} ^$
RewriteCond %{REQUEST_URI} !checkout [NC]
RewriteCond %{REQUEST_URI} !^/kasse [NC]
RewriteCond %{REQUEST_URI} !login [NC]
RewriteCond %{REQUEST_URI} !account [NC]
RewriteCond %{REQUEST_URI} !shopping_cart [NC]
RewriteCond %{REQUEST_URI} !^/warenkorb [NC]
RewriteCond %{REQUEST_URI} !logoff [NC]
RewriteCond %{REQUEST_URI} !^/vergleich [NC]
RewriteCond %{REQUEST_URI} !^/wishlist [NC]
RewriteCond %{REQUEST_URI} !password_double_opt [NC]
RewriteCond %{REQUEST_URI} !create_account [NC]
RewriteCond %{REQUEST_URI} !contact_us [NC]
RewriteCond %{REQUEST_URI} !tell_a_friend [NC]
RewriteCond %{REQUEST_URI} !product_reviews_write [NC]
RewriteCond /home/www/doc/28856/dcp288560004/mr-hanf.de/www/cache/fpc/%{REQUEST_URI}/index.html -f
RewriteRule ^(.+)$ fpc_serve.php [L,QSA,E=FPC_HIT:1]
</IfModule>

# === FPC Cache-Control Header (via Environment-Variable) ===
# fpc_serve.php setzt die Header selbst per PHP header() Funktion.
# Diese Regeln hier sind ein zusaetzlicher Fallback.
<IfModule mod_headers.c>
    Header always set X-FPC-Cache "HIT" env=REDIRECT_FPC_HIT
    Header always set X-FPC-Version "8.0.7" env=REDIRECT_FPC_HIT
    Header always set Cache-Control "no-store, no-cache, must-revalidate" env=REDIRECT_FPC_HIT
    Header always set Pragma "no-cache" env=REDIRECT_FPC_HIT
</IfModule>
# --- FPC ENDE v8.0.7 ---
